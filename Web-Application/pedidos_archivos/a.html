                                                                      var db = require('./queries');

                                                                        res.send('OK!');
                                                                      });

                                                                      var db = require('./queries');
                                                                      app.get('/api/puppies', db.getAllPuppies);

                                                                      var pgp = require('pg-promise')(options);
                                                                      var db = pgp({
                                                                          host: 'localhost',
                                                                          port: 5432,
                                                                          database: 'puppies',
                                                                          user: 'postgres',
                                                                          password: 'mypassword'
                                                                      });
